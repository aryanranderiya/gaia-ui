import { NavbarMenu } from "@/components/ui/navbar-menu";

export default function Example() {

        {

### Full Navbar Implementation

This is a complete example showing how to integrate the menu into a navbar with hover interactions:

```tsx
"use client";

import { useState } from "react";
import { AnimatePresence } from "motion/react";
import { ChevronDown } from "lucide-react";
import { NavbarMenu } from "@/components/ui/navbar-menu";
import { cn } from "@/lib/utils";

export default function Navbar() {
  const [activeDropdown, setActiveDropdown] = useState<string | null>(null);
  const [hoveredItem, setHoveredItem] = useState<string | null>(null);

  const navItems = [
    { type: "dropdown", label: "Product", menu: "product" },
    { type: "dropdown", label: "Resources", menu: "resources" },
    { type: "link", label: "Pricing", href: "/pricing" },
    { type: "dropdown", label: "Company", menu: "company" },
  ] as const;

  const sections = [
    {
      id: "product",
      links: [
        {
          label: "Features",
          href: "/features",
          description: "Explore our powerful features",
          icon: <FeatureIcon />,
        },
        // ... more links
      ],
    },
    // ... more sections
  ];

  const handleNavbarMouseLeave = () => {
    setActiveDropdown(null);
    setHoveredItem(null);
  };

  const handleMouseEnter = (menu: string) => {
    setActiveDropdown(menu);
    setHoveredItem(menu);
  };

  return (
    <div
      className="relative mx-auto max-w-6xl"
      onMouseLeave={handleNavbarMouseLeave}
    >
      <div
        className={cn(
          "flex h-14 w-full items-center justify-between border border-white/5 px-3 backdrop-blur-md transition-all",
          activeDropdown
            ? "rounded-t-2xl border-b-0 bg-zinc-950"
            : "rounded-2xl bg-zinc-900/30"
        )}
      >
        {/* Logo */}
        <div className="flex items-center gap-2 px-2">
          <Logo />
        </div>

        {/* Navigation Items */}
        <div className="flex items-center gap-1">
          {navItems.map((item) =>
            item.type === "link" ? (
              <a
                key={item.href}
                href={item.href}
                className="flex h-9 items-center rounded-xl px-4 py-2 text-sm text-zinc-400 hover:text-zinc-100"
                onMouseEnter={() => {
                  setHoveredItem(item.label.toLowerCase());
                  setActiveDropdown(null);
                }}
              >
                {item.label}
              </a>
            ) : (
              <button
                key={item.menu}
                className="relative flex h-9 items-center rounded-xl px-4 py-2 text-sm text-zinc-400 hover:text-zinc-100"
                onMouseEnter={() => handleMouseEnter(item.menu)}
              >
                {hoveredItem === item.menu && (
                  <div className="absolute inset-0 rounded-xl bg-zinc-800" />
                )}
                <span className="relative z-10 flex items-center gap-2">
                  {item.label}
                  <ChevronDown
                    className={cn(
                      "h-4 w-4 transition",
                      hoveredItem === item.menu && "rotate-180"
                    )}
                  />
                </span>
              </button>
            )
          )}
        </div>

        {/* CTA Buttons */}
        <div className="flex items-center gap-2">
          <button className="rounded-xl bg-zinc-800 px-4 py-2 text-sm">
            Sign In
          </button>
          <button className="rounded-xl bg-blue-500 px-4 py-2 text-sm">
            Get Started
          </button>
        </div>
      </div>

      {/* Dropdown Menu */}
      <AnimatePresence>
        {activeDropdown && (
          <NavbarMenu
            activeMenu={activeDropdown}
            sections={sections}
            onClose={() => setActiveDropdown(null)}
          />
        )}
      </AnimatePresence>
    </div>
  );
}
```

### Key Implementation Details

1. **State Management**: Use two state variables - `activeDropdown` for the current menu and `hoveredItem` for visual feedback
2. **Mouse Events**: Handle `onMouseEnter` for opening menus and `onMouseLeave` on the navbar container to close them
3. **Conditional Styling**: Change navbar border radius and background when menu is open for a seamless appearance
4. **AnimatePresence**: Wrap the NavbarMenu with AnimatePresence for smooth exit animations

## Props

### NavbarMenu

| Prop       | Type                | Default   | Description                                 |
| ---------- | ------------------- | --------- | ------------------------------------------- |
| activeMenu | string              | -         | The ID of the currently active menu section |
| sections   | NavbarMenuSection[] | -         | Array of menu sections with links           |
| onClose    | () => void          | undefined | Optional callback when menu should close    |

### NavbarMenuSection

| Property   | Type             | Description                                              |
| ---------- | ---------------- | -------------------------------------------------------- |
| id         | string           | Unique identifier for the section                        |
| links      | NavbarMenuLink[] | Array of links in this section                           |
| gridLayout | string           | Optional custom grid layout classes (default: 2 columns) |

### NavbarMenuLink

| Property        | Type            | Description                                        |
| --------------- | --------------- | -------------------------------------------------- |
| label           | string          | Link text                                          |
| href            | string          | Link URL                                           |
| icon            | React.ReactNode | Optional icon component                            |
| external        | boolean         | Whether link opens in new tab                      |
| description     | string          | Optional description text                          |
| backgroundImage | string          | Optional background image URL for featured items   |
| rowSpan         | number          | Optional row span for grid layout (2 for featured) |

## Features

- **Smooth Animations**: Uses Framer Motion for buttery smooth enter/exit animations
- **Flexible Grid Layouts**: Customize grid layouts per section (2-col, 3-col, etc.)
- **Featured Items**: Support for large featured items with background images
- **Icon Support**: Add icons to any menu item
- **External Links**: Automatically handles external links with proper attributes
- **Responsive**: Works on all screen sizes
- **Accessible**: Keyboard navigation and ARIA attributes
